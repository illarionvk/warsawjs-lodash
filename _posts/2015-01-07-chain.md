## _.chain()

{% highlight javascript linenos %}
_.chain([1, 2, 3, 4, 5, 6])
  .filter(function(value) { return value > 3; }) // → [4, 5, 6]
  .map(function(value) { return value * 5; }) // → [20, 25, 30]
  .reduce(function(accumulator, value) {
    return accumulator + value;
  }, 0)
  .value();
// → 75
{% endhighlight %}
